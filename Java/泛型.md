为什么数组不支持泛型

    //Java允许下面的代码编译正常
    //但是运行的时候会抛出ArrayStoreException异常
    String[] sArray = new String[4];
    Object[] oArray = sArray;
    oArray[0] = new Integer(1);
    
    //现在假设数组允许泛型
    //由于类型擦除的影响
    //就会导致下面的语句运行的时候并不会抛出异常
    Generic<Integer>[] gArray = new Generic<Integer>[4];
    Object[] oArray = gArray;
    oArray[0] = new Generic<String>();
    //下面的方式就会抛出异常
    //这种行为是不安全的,属于低级别的错误,所以Java不允许泛型数组
    Integer val = gArray[0].getVal();
        
    但是数组可以使用泛型通配符
    List<?>[] list = new ArrayList<?>[4];
    list[0] = new ArrayList<String>();
    list[0].get(0);
    list[0].size();

为什么泛型会有通配符
    
```java

    //假设有个打印的方法print
    //这个方法的参数类型是List<Number>
    void print(List<Number> list) {
    }

    //此时如果我想复用上面的方法,也可以打印Number的子类Integer
    List<Integer> il = new ArrayList<>();
    //这行代码就会编译不通过
    print(il);

```

为了解决上面的问题通配符就出现了  
三种通配符

1.<?>
```java

    print(List<?> list) {
        //只能调用List中与类型无关的方法
        list.size();
        //下面编译不通过
        //至于为什么不通过,可以参考下面的内容
        //list.add("cr");
    }

```
    
2.<? extend XXX>
```java

    //? extends Number表示Number或者其子类
    List<? extends Number> list = new ArrayList<Integer>();
    
    //可以使用List的get方法
    list.get(0).byteValue();
    //可以添加null
    list.add(null);
    
    //编译不通过
    list.add(1);
    为什么不能add(1)?
    //假设可以添加,那么下面的方法内部就可以往list中添加Integer,Float
    //如果调用方法使用print(new ArrayList<Integer>())
    print(List<? extends Number> list) {
        list.add(27);
        //此处抛异常
        list.add(2.7F);
    }

```
    
3.<? super XXX>
```java

     //?表示Numer或者其父类
     List<? super Number> list = new ArrayList<Object>(); 
     list.add(null);
     list.add(1);
     
     为什么这里可以add(1)?
     //通过方法签名不确定哪个Number的父类传入进来
     //所以不能添加Number的父类
     //但是可以确定Number的子类一定是继承Number的父类
     //所以可以添加Number的子类
     print(List<? super Number> list) {
        list.add(2.7F);
        list.add(27);
    }
    
```

泛型类和泛型方法中的T没有关系

    public class GenericClassAndGenericMethod<T> {
        public void test1(T t) {
            System.out.println(t.getClass().getName());
        }
        public <T> T test2(T t) {
            return t;
        }
        public static void main(String[] args) {
            GenericClassAndGenericMethod<String> t = new GenericClassAndGenericMethod();
            //这里必须传入String类型
            t.test1("CR");
            //这里可以传入任意类型
            t.test2(new Integer(1));
        }
    }

类型擦除

    public class Generic<T> {
        private T t;
        public Generic(T t) {
            this.t = t;
        }
        public void test(T t) {
            System.out.println(t);
        }
    }
    Generic<String> g = new Generic<>("cr");
    g.getClass().getName(); //com.cr.generic.Generic
    Field[] fields = g.getClass().getDeclaredFields();
    for (Field field : fields) {
        System.out.println(field.getType().getName()); //java.lang.Object
    }
    Method[] methods = g.getClass().getDeclaredMethods();
    for (Method method : methods) {
        System.out.println(method); //public void com.cr.generic.Generic.test(java.lang.Object)
    }

    public class Generic<T extends Number> {
        private T t;
    }
    Generic<Integer> g = new Generic2<>();
    Field[] fields = g.getClass().getDeclaredFields();
    for (Field field : fields) {
        //这里打印不是Object
        //java.lang.Number
        System.out.println(field.getType().getName());
    }

    List<Integer> list = new ArrayList<>();
    Method add = list.getClass().getDeclaredMethod("add", Object.class);
    add.invoke(list, "c");
    add.invoke(list, "r");
    for (Object o : list) {
        System.out.println(o);
    }
