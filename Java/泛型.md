为什么数组不支持泛型

    //Java允许下面的代码编译正常,但是运行的时候抛出ArrayStoreException异常
    String[] sArray = new String[4];
    Object[] oArray = sArray;
    oArray[0] = new Integer(1);
    
    //现在假设数组允许泛型
    //类型擦除的影响,下面的语句运行的时候就会变成合法的了
    Generic<Integer>[] gArray = new Generic<Integer>[4];
    Object[] oArray = gArray;
    oArray[0] = new Generic<String>();

为什么泛型会有通配符
    
    //假设我有个打印的方法print
    //这个方法的参数是List集合
    //这个集合所有的类型都是Number
    void print(List<Number> list){
    }
    
    //如果我想复用这个方法
    //也可以打印Number的子类
    List<Integer> iList = new ArrayList<>();
    //这行代码就会编译不通过
    print(iList);
    
    为了解决上面的问题通配符就出现了

三种通配符

1.<?>

    print(List<?> list) {
        //print方法内只能调用List类中与类型无关的方法
        list.size();
        //下面编译不通过
        //list.add("cr");
    }
    
2.<? extend XXX>

    //?表示Number或者其子类
    List<? extends Number> list = new ArrayList<Integer>();
    
    //可以使用List的get方法
    list.get(0).byteValue();
    
    list.add(null);
    
    //编译不通过
    list.add(1);
    
    为什么不能add(1)?
    //下面通过方法签名我可以添加Integer,Float
    //如果调用方法print(new ArrayList<Integer>())
    //添加float就会报错
    //所以不允许添加元素除了null
    print(List<? extends Number> list) {
        list.add(2.7F);
        list.add(27);
    }
    
3.<? super XXX>

     //?表示Numer或者其父类
     List<? super Number> list = new ArrayList<Object>(); 
     list.add(null);
     list.add(1);
     
     为什么这里可以add(1)?
     //通过方法签名不确定哪个Number的父类传入进来
     //所以不能添加Number的父类
     //但是可以确定Number的子类一定是继承Number的父类
     //所以可以添加Number的子类
     print(List<? super Number> list) {
        list.add(2.7F);
        list.add(27);
    }
